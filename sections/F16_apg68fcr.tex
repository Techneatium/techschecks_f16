\subsection{RWS}
\label{subsec:rws}
\begin{tcoloritemize}
    \blueitem{RWS}{
    \begin{subitemize}
        \item \textbf{R}ange \textbf{W}hile \textbf{S}earch
        \item \textbf{Fast, Long(er)-Range Search}
    \end{subitemize}}
    \blueitem{SAM \break Submode}{
    \begin{subitemize}
        \item \textbf{S}ituational \textbf{A}warness \textbf{M}ode
        \item \textbf{Target is ``Bugged''} (Pseudo-Track)
        \begin{itemize}
            \item Can guide AIM-120C (w/o STT Lock)
        \end{itemize}
    \end{subitemize}

    SAM Acquisition

    \begin{subenumerate}
        \item \textbf{Target} \dotfill under Acquisition Cursor
        \item \textbf{TMS} \dotfill \textbf{Forward (hold)}
        \begin{itemize}
            \item \textbf{Scan Pattern} -- 4-bar, \pm 10$^\circ$ spotlight
        \end{itemize}
        \item \textbf{Target} \dotfill Verify \textbf{Bugged}
        \begin{itemize}
            \item Can guide AIM-120C (w/o STT Lock)
            \item DLZ displayed if missile selected
        \end{itemize}
    \end{subenumerate}
    
    Exit SAM 

    \begin{subenumerate}
        \item \textbf{TMS} \dotfill \textbf{Aft}
    \end{subenumerate}}
    \blueitem{DTT \break Submode}{
    \begin{subitemize}
        \item \textbf{D}ual \textbf{T}arget \textbf{T}rack
        \item \textbf{2 Targets ``Bugged''} -- primary/secondary
        \begin{itemize}
            \item Can guide 120C on primary target
        \end{itemize}
    \end{subitemize}

    DTT Acquisition

    \begin{subenumerate}
        \item \textbf{Target 1} \dotfill \textbf{Bugged} in SAM
        \item \textbf{Target 2} \dotfill under Acquisition Cursor
        \item \textbf{TMS} \dotfill \textbf{Forward (hold)} to bug
    \end{subenumerate}
    
    Exit SAM 
    
    \begin{subenumerate}
        \item \textbf{TMS} \dotfill \textbf{Aft}
    \end{subenumerate}}
\end{tcoloritemize}

\subsection{TWS}
\label{subsec:tws}
\begin{tcoloritemize}
    \blueitem{TWS}{
    \begin{subitemize}
        \item \textbf{T}rack \textbf{W}hile \textbf{S}can
        \item \textbf{Multi-target tracking mode}
        \item \textbf{FCR builds trackfiles for each target}
        \item \textbf{Trackfiles can be in several states} --
        Search / Track / System / Cursor / Bugged
    \end{subitemize}}
    \blueitem{Search Target}{
    \begin{subitemize}
        \item Initial state 
        \item Not enough information to become track
        \item Transition to track after 1-2 radar sweeps
        \item \textbf{Display}
        \begin{itemize}
            \item like RWS radar brick
        \end{itemize} 
    \end{subitemize}}
    \blueitem{Track Target}{
    \begin{subitemize}
        \item Once trackfile generated
        \item Can be transitioned to higher fidelity modes by pilot
        \item \textbf{Display}
        \begin{itemize}
            \item larger brick with velocity vector
            \item numerical altitude to below brick
            \item max 10 displayed
        \end{itemize} 
    \end{subitemize}}
    \blueitem{System Target}{
    \begin{subitemize}
        \item \textbf{Upgrade:} 
        \begin{itemize}
            \item \textbf{TMS Fwd} with cursor over Track Target
            \item \textbf{TMS Right} to upgrade all Track Targets to System Targets
        \end{itemize}
        \item \textbf{Scan} -- No additional radar energy
        \item \textbf{Downgrade} -- \textbf{TMS Aft} to downgrade to Track Target
    \end{subitemize}}
    \blueitem{Cursor Target}{
    \begin{subitemize}
        \item \textbf{Upgrade} -- position cursor over System Target
        \item \textbf{Cycle} -- slew to other System Target
        \item \textbf{Scan}
        \begin{itemize}
            \item 3-bar, $\pm$25 deg pattern 
            \item centered on Cursor Target
        \end{itemize}
        \item \textbf{Downgrade} -- slew away from all System Targets
    \end{subitemize}}
    \blueitem{Bugged Target}{
    \begin{subitemize}
        \item \textbf{Upgrade:} 
        \begin{itemize}
            \item \textbf{TMS Fwd} with cursor over system target to designate
            \item \textbf{TMS Right} to select closest System Target 
        \end{itemize}
        \item \textbf{Cycle} -- \textbf{TMS Right} to cycle in range order
        \item \textbf{Scan}
        \begin{itemize}
            \item 3-bar, $\pm$25 deg pattern 
            \item centered on bugged target
        \end{itemize}
        \item \textbf{Weapons}
        \begin{itemize}
            \item Target selected for AIM-9 / -120 employment
            \item DLZ information displayed on HUD/FCR
        \end{itemize}
        \item \textbf{Downgrade} -- \textbf{TMS Aft} to downgrade to System Target
    \end{subitemize}}
\end{tcoloritemize}
